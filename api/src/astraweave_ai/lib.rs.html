7/// Minimal rule-based orchestrator:
8/// If enemy in LOS-ish and "smoke" not on cooldown:
9///   throw smoke midway, move up, cover fire.
10/// Else: advance towards nearest enemy.
11pub struct RuleOrchestrator;
12
13impl Orchestrator for RuleOrchestrator {
14    fn propose_plan(&self, snap: &WorldSnapshot) -> PlanIntent {
15        let plan_id = format!("plan-{}", (snap.t * 1000.0) as i64);
16        if let Some(first) = snap.enemies.first() {
17            let m = &snap.me;
18            let _p = &snap.player;
19
20            // midpoint for smoke
21            let mid = IVec2 {
22                x: (m.pos.x + first.pos.x) / 2,
23                y: (m.pos.y + first.pos.y) / 2,
24            };
25
26            let smoke_cd = snap.me.cooldowns.get("throw:smoke").copied().unwrap_or(0.0);
27            if smoke_cd <= 0.0 {
28                return PlanIntent {
29                    plan_id,
30                    steps: vec![
31                        ActionStep::Throw {
32                            item: "smoke".into(),
33                            x: mid.x,
34                            y: mid.y,
35                        },
36                        ActionStep::MoveTo {
37                            x: m.pos.x + (first.pos.x - m.pos.x).signum() * 2,
38                            y: m.pos.y + (first.pos.y - m.pos.y).signum() * 2,
39                        },
40                        ActionStep::CoverFire {
41                            target_id: first.id,
42                            duration: 2.5,
43                        },
44                    ],
45                };
46            } else {
47                // advance cautiously
48                return PlanIntent {
49                    plan_id,
50                    steps: vec![
51                        ActionStep::MoveTo {
52                            x: m.pos.x + (first.pos.x - m.pos.x).signum(),
53                            y: m.pos.y + (first.pos.y - m.pos.y).signum(),
54                        },
55                        ActionStep::CoverFire {
56                            target_id: first.id,
57                            duration: 1.5,
58                        },
59                    ],
60                };
61            }
62        }
63
64        // fallback
65        PlanIntent {
66            plan_id,
67            steps: vec![],
68        }
69    }
70}
