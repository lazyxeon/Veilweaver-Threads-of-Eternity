15const SHADER: &str = r#"
16struct Params { fade: f32, letterbox: f32, _pad: vec2<f32> };
17@group(0) @binding(0) var<uniform> U: Params;
18
19struct VSOut { @builtin(position) pos: vec4<f32>, @location(0) ndc: vec2<f32> };
20@vertex
21fn vs_main(@builtin(vertex_index) vid: u32) -> VSOut {
22  var pos = array<vec2<f32>, 3>(
23    vec2<f32>(-1.0, -3.0), vec2<f32>(3.0, -1.0), vec2<f32>(-1.0, 1.0)
24  );
25  var out: VSOut;
26  out.pos = vec4<f32>(pos[vid], 0.0, 1.0);
27  out.ndc = pos[vid];
28  return out;
29}
30@fragment
31fn fs_main(inf: VSOut) -> @location(0) vec4<f32> {
32  let y = (inf.ndc.y * 0.5 + 0.5);
33  let lb = U.letterbox;
34  var col = vec4<f32>(0.0, 0.0, 0.0, U.fade);
35  if (y < lb || y > (1.0 - lb)) {
36    // letterbox bar: opaque black
37    col = vec4<f32>(0.0, 0.0, 0.0, 1.0);
38  }
39  return col;
40}
41"#;

43impl OverlayFx {
44    pub fn new(device: &wgpu::Device, format: wgpu::TextureFormat) -> Self {
45        let buf = device.create_buffer(&wgpu::BufferDescriptor {
46            label: Some("overlay params"),
47            size: std::mem::size_of::<OverlayParams>() as u64,
48            usage: wgpu::BufferUsages::UNIFORM | wgpu::BufferUsages::COPY_DST,
49            mapped_at_creation: false,
50        });
51        let bgl = device.create_bind_group_layout(&wgpu::BindGroupLayoutDescriptor {
52            label: Some("overlay bgl"),
53            entries: &[wgpu::BindGroupLayoutEntry {
54                binding: 0,
55                visibility: wgpu::ShaderStages::VERTEX_FRAGMENT,
56                ty: wgpu::BindingType::Buffer {
57                    ty: wgpu::BufferBindingType::Uniform,
58                    has_dynamic_offset: false,
59                    min_binding_size: None,
60                },
61                count: None,
62            }],
63        });
64        let bind = device.create_bind_group(&wgpu::BindGroupDescriptor {
65            label: Some("overlay bind"),
66            layout: &bgl,
67            entries: &[wgpu::BindGroupEntry {
68                binding: 0,
69                resource: buf.as_entire_binding(),
70            }],
71        });
72        let shader = device.create_shader_module(wgpu::ShaderModuleDescriptor {
73            label: Some("overlay shader"),
74            source: wgpu::ShaderSource::Wgsl(SHADER.into()),
75        });
76        let pl = device.create_pipeline_layout(&wgpu::PipelineLayoutDescriptor {
77            label: Some("overlay pl"),
78            bind_group_layouts: &[&bgl],
79            push_constant_ranges: &[],
80        });
81        let pipeline = device.create_render_pipeline(&wgpu::RenderPipelineDescriptor {
82            label: Some("overlay pipe"),
83            layout: Some(&pl),
84            vertex: wgpu::VertexState {
85                module: &shader,
86                entry_point: "vs_main",
87                buffers: &[],
88                compilation_options: wgpu::PipelineCompilationOptions::default(),
89            },
90            fragment: Some(wgpu::FragmentState {
91                module: &shader,
92                entry_point: "fs_main",
93                targets: &[Some(wgpu::ColorTargetState {
94                    format,
95                    blend: Some(wgpu::BlendState::ALPHA_BLENDING),
96                    write_mask: wgpu::ColorWrites::ALL,
97                })],
98                compilation_options: wgpu::PipelineCompilationOptions::default(),
99            }),
100            primitive: wgpu::PrimitiveState::default(),
101            depth_stencil: None,
102            multisample: wgpu::MultisampleState::default(),
103            multiview: None,
104        });
105        Self {
106            buf,
107            pipeline,
108            bind,
109        }
110    }
111
112    pub fn update(&self, queue: &wgpu::Queue, p: &OverlayParams) {
113        queue.write_buffer(&self.buf, 0, bytemuck::bytes_of(p));
114    }
115
116    pub fn draw<'a>(&'a self, rpass: &mut wgpu::RenderPass<'a>) {
117        rpass.set_pipeline(&self.pipeline);
118        rpass.set_bind_group(0, &self.bind, &[]);
119        rpass.draw(0..3, 0..1);
120    }
