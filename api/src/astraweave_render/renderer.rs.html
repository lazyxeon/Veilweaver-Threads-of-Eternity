11const SHADER_SRC: &str = r#"
12struct VSIn {
13  @location(0) position: vec3<f32>,
14  @location(1) normal:   vec3<f32>,
15  @location(2) m0: vec4<f32>,
16  @location(3) m1: vec4<f32>,
17  @location(4) m2: vec4<f32>,
18  @location(5) m3: vec4<f32>,
19  @location(6) n0: vec3<f32>,
20  @location(7) n1: vec3<f32>,
21  @location(8) n2: vec3<f32>,
22  @location(9) color: vec4<f32>,
23};
24
25struct VSOut {
26  @builtin(position) pos: vec4<f32>,
27  @location(0) world_pos: vec3<f32>,
28  @location(1) normal: vec3<f32>,
29  @location(2) color: vec4<f32>,
30};
31
32struct Camera {
33  view_proj: mat4x4<f32>,
34  light_dir: vec3<f32>,
35  _pad: f32,
36};
37
38@group(0) @binding(0) var<uniform> uCamera: Camera;
39
40@vertex
41fn vs(input: VSIn) -> VSOut {
42  let model = mat4x4<f32>(input.m0, input.m1, input.m2, input.m3);
43  let world = model * vec4<f32>(input.position, 1.0);
44  var out: VSOut;
45  out.pos = uCamera.view_proj * world;
46  // normal matrix simplified (assuming uniform scale); for accuracy pass and use 3x3
47  out.normal = normalize((model * vec4<f32>(input.normal, 0.0)).xyz);
48  out.world_pos = world.xyz;
49  out.color = input.color;
50  return out;
51}
52
53@fragment
54fn fs(input: VSOut) -> @location(0) vec4<f32> {
55  let n = normalize(input.normal);
56  let l = normalize(-uCamera.light_dir);
57  let ndl = max(dot(n, l), 0.0);
58  let ambient = 0.2;
59  let diff = ndl * 0.8;
60  let lit = (ambient + diff);
61  let shaded = vec4<f32>(input.color.rgb * lit, input.color.a);
62  return shaded;
63}
64"#;

101    pub async fn new(window: std::sync::Arc<winit::window::Window>) -> Result<Self> {
102        // WGPU init
103        let instance = wgpu::Instance::new(wgpu::InstanceDescriptor::default());
104        let surface = instance.create_surface(window.clone())?;
105        let adapter = instance
106            .request_adapter(&wgpu::RequestAdapterOptions {
107                power_preference: wgpu::PowerPreference::HighPerformance,
108                compatible_surface: Some(&surface),
109                force_fallback_adapter: false,
110            })
111            .await
112            .ok_or_else(|| anyhow::anyhow!("No adapter"))?;
113
114        let (device, queue) = adapter
115            .request_device(
116                &wgpu::DeviceDescriptor {
117                    label: Some("device"),
118                    required_features: wgpu::Features::empty(),
119                    required_limits: wgpu::Limits::default(),
120                },
121                None,
122            )
123            .await?;
124
125        // Surface config
126        let caps = surface.get_capabilities(&adapter);
127        let format = caps
128            .formats
129            .iter()
130            .copied()
131            .find(|f| f.is_srgb())
132            .unwrap_or(caps.formats[0]);
133        let size = window.inner_size();
134        let config = wgpu::SurfaceConfiguration {
135            usage: wgpu::TextureUsages::RENDER_ATTACHMENT,
136            format,
137            width: size.width.max(1),
138            height: size.height.max(1),
139            present_mode: caps.present_modes[0],
140            alpha_mode: caps.alpha_modes[0],
141            view_formats: vec![],
142            desired_maximum_frame_latency: 2,
143        };
144        surface.configure(&device, &config);
145
146        // Depth
147        let depth = crate::depth::Depth::create(&device, &config);
148
149        // Shaders / pipeline
150        let shader = device.create_shader_module(wgpu::ShaderModuleDescriptor {
151            label: Some("basic shader"),
152            source: wgpu::ShaderSource::Wgsl(Cow::Borrowed(SHADER_SRC)),
153        });
154
155        let camera_buf = device.create_buffer(&wgpu::BufferDescriptor {
156            label: Some("camera ubo"),
157            size: std::mem::size_of::<CameraUBO>() as u64,
158            usage: wgpu::BufferUsages::UNIFORM | wgpu::BufferUsages::COPY_DST,
159            mapped_at_creation: false,
160        });
161
162        let bind_layout = device.create_bind_group_layout(&wgpu::BindGroupLayoutDescriptor {
163            label: Some("camera bgl"),
164            entries: &[wgpu::BindGroupLayoutEntry {
165                binding: 0,
166                visibility: wgpu::ShaderStages::VERTEX_FRAGMENT,
167                ty: wgpu::BindingType::Buffer {
168                    ty: wgpu::BufferBindingType::Uniform,
169                    has_dynamic_offset: false,
170                    min_binding_size: None,
171                },
172                count: None,
173            }],
174        });
175
176        let camera_bind_group = device.create_bind_group(&wgpu::BindGroupDescriptor {
177            label: Some("camera bg"),
178            layout: &bind_layout,
179            entries: &[wgpu::BindGroupEntry {
180                binding: 0,
181                resource: camera_buf.as_entire_binding(),
182            }],
183        });
184
185        let pipeline_layout = device.create_pipeline_layout(&wgpu::PipelineLayoutDescriptor {
186            label: Some("pipeline layout"),
187            bind_group_layouts: &[&bind_layout],
188            push_constant_ranges: &[],
189        });
190
191        let pipeline = device.create_render_pipeline(&wgpu::RenderPipelineDescriptor {
192            label: Some("pipeline"),
193            layout: Some(&pipeline_layout),
194            vertex: wgpu::VertexState {
195                module: &shader,
196                entry_point: "vs",
197                buffers: &[
198                    crate::types::Vertex::layout(),
199                    crate::types::InstanceRaw::layout(),
200                ],
201                compilation_options: wgpu::PipelineCompilationOptions::default(),
202            },
203            fragment: Some(wgpu::FragmentState {
204                module: &shader,
205                entry_point: "fs",
206                targets: &[Some(wgpu::ColorTargetState {
207                    format,
208                    blend: Some(wgpu::BlendState::ALPHA_BLENDING),
209                    write_mask: wgpu::ColorWrites::ALL,
210                })],
211                compilation_options: wgpu::PipelineCompilationOptions::default(),
212            }),
213            primitive: wgpu::PrimitiveState {
214                topology: wgpu::PrimitiveTopology::TriangleList,
215                strip_index_format: None,
216                front_face: wgpu::FrontFace::Ccw,
217                cull_mode: Some(wgpu::Face::Back),
218                polygon_mode: wgpu::PolygonMode::Fill,
219                conservative: false,
220                unclipped_depth: false,
221            },
222            depth_stencil: Some(wgpu::DepthStencilState {
223                format: depth.format,
224                depth_write_enabled: true,
225                depth_compare: wgpu::CompareFunction::Less,
226                stencil: wgpu::StencilState::default(),
227                bias: wgpu::DepthBiasState::default(),
228            }),
229            multisample: wgpu::MultisampleState::default(),
230            multiview: None,
231        });
232
233        let overlay = crate::overlay::OverlayFx::new(&device, format);
234
235        let overlay_params = crate::overlay::OverlayParams {
236            fade: 0.0,
237            letterbox: 0.0,
238            _pad: [0.0; 2],
239        };
240
241        // Meshes
242        let (cube_v, cube_i) = primitives::cube();
243        let mesh_cube = Mesh {
244            vertex_buf: device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
245                label: Some("cube v"),
246                contents: bytemuck::cast_slice(&cube_v),
247                usage: wgpu::BufferUsages::VERTEX,
248            }),
249            index_buf: device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
250                label: Some("cube i"),
251                contents: bytemuck::cast_slice(&cube_i),
252                usage: wgpu::BufferUsages::INDEX,
253            }),
254            index_count: cube_i.len() as u32,
255        };
256
257        let (plane_v, plane_i) = primitives::plane();
258        let mesh_plane = Mesh {
259            vertex_buf: device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
260                label: Some("plane v"),
261                contents: bytemuck::cast_slice(&plane_v),
262                usage: wgpu::BufferUsages::VERTEX,
263            }),
264            index_buf: device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
265                label: Some("plane i"),
266                contents: bytemuck::cast_slice(&plane_i),
267                usage: wgpu::BufferUsages::INDEX,
268            }),
269            index_count: plane_i.len() as u32,
270        };
271
272        // Instance buffer (start empty; caller will fill)
273        let instance_buf = device.create_buffer(&wgpu::BufferDescriptor {
274            label: Some("instance buf"),
275            size: 1024 * 1024, // 1MB to start (grow if needed)
276            usage: wgpu::BufferUsages::VERTEX | wgpu::BufferUsages::COPY_DST,
277            mapped_at_creation: false,
278        });
279
280        // after instance_buf creation
281        let weather = crate::effects::WeatherFx::new(&device, 800);
282
283        Ok(Self {
284            surface,
285            device,
286            queue,
287            config,
288            depth,
289            shader,
290            pipeline,
291            camera_ubo: CameraUBO {
292                view_proj: Mat4::IDENTITY.to_cols_array_2d(),
293                light_dir_pad: [0.5, 1.0, 0.8, 0.0],
294            },
295            camera_buf,
296            camera_bind_group,
297            mesh_cube,
298            mesh_plane,
299            instances: Vec::new(),
300            instance_buf,
301            overlay,
302            overlay_params,
303            weather,
304        })
305    }

307    pub fn resize(&mut self, new_w: u32, new_h: u32) {
308        if new_w == 0 || new_h == 0 {
309            return;
310        }
311        self.config.width = new_w;
312        self.config.height = new_h;
313        self.surface.configure(&self.device, &self.config);
314        self.depth = crate::depth::Depth::create(&self.device, &self.config);
315    }

317    pub fn update_camera(&mut self, camera: &Camera) {
318        self.camera_ubo.view_proj = camera.vp().to_cols_array_2d();
319        self.queue
320            .write_buffer(&self.camera_buf, 0, bytemuck::bytes_of(&self.camera_ubo));
321    }

323    pub fn update_instances(&mut self, instances: &[Instance]) {
324        self.instances.clear();
325        self.instances.extend_from_slice(instances);
326        let raws: Vec<InstanceRaw> = self.instances.iter().map(|i| i.raw()).collect();
327        let size = (raws.len() * std::mem::size_of::<InstanceRaw>()) as u64;
328
329        if size > self.instance_buf.size() {
330            self.instance_buf = self.device.create_buffer(&wgpu::BufferDescriptor {
331                label: Some("instance buf (resized)"),
332                size: size.next_power_of_two(),
333                usage: wgpu::BufferUsages::VERTEX | wgpu::BufferUsages::COPY_DST,
334                mapped_at_creation: false,
335            });
336        }
337        self.queue
338            .write_buffer(&self.instance_buf, 0, bytemuck::cast_slice(&raws));
339    }

349    pub fn render(&mut self) -> Result<()> {
350        let frame = self.surface.get_current_texture()?;
351        let view = frame
352            .texture
353            .create_view(&wgpu::TextureViewDescriptor::default());
354
355        let mut enc = self
356            .device
357            .create_command_encoder(&wgpu::CommandEncoderDescriptor {
358                label: Some("encoder"),
359            });
360
361        // Create plane buffer before render pass
362        let plane_scale = glam::Mat4::from_scale(vec3(50.0, 1.0, 50.0));
363        let plane_inst = Instance {
364            transform: plane_scale,
365            color: [0.1, 0.12, 0.14, 1.0],
366        }
367        .raw();
368        let plane_buf = self
369            .device
370            .create_buffer_init(&wgpu::util::BufferInitDescriptor {
371                label: Some("plane inst"),
372                contents: bytemuck::bytes_of(&plane_inst),
373                usage: wgpu::BufferUsages::VERTEX,
374            });
375
376        {
377            let mut rp = enc.begin_render_pass(&wgpu::RenderPassDescriptor {
378                label: Some("main pass"),
379                color_attachments: &[Some(wgpu::RenderPassColorAttachment {
380                    view: &view,
381                    resolve_target: None,
382                    ops: wgpu::Operations {
383                        load: wgpu::LoadOp::Clear(wgpu::Color {
384                            r: 0.05,
385                            g: 0.07,
386                            b: 0.10,
387                            a: 1.0,
388                        }),
389                        store: wgpu::StoreOp::Store,
390                    },
391                })],
392                depth_stencil_attachment: Some(wgpu::RenderPassDepthStencilAttachment {
393                    view: &self.depth.view,
394                    depth_ops: Some(wgpu::Operations {
395                        load: wgpu::LoadOp::Clear(1.0),
396                        store: wgpu::StoreOp::Store,
397                    }),
398                    stencil_ops: None,
399                }),
400                timestamp_writes: None,
401                occlusion_query_set: None,
402            });
403
404            rp.set_pipeline(&self.pipeline);
405            rp.set_bind_group(0, &self.camera_bind_group, &[]);
406
407            // Ground plane (scaled)
408            rp.set_vertex_buffer(0, self.mesh_plane.vertex_buf.slice(..));
409            rp.set_index_buffer(
410                self.mesh_plane.index_buf.slice(..),
411                wgpu::IndexFormat::Uint32,
412            );
413            rp.set_vertex_buffer(1, plane_buf.slice(..));
414            rp.draw_indexed(0..self.mesh_plane.index_count, 0, 0..1);
415
416            // Cubes (instances)
417            rp.set_vertex_buffer(0, self.mesh_cube.vertex_buf.slice(..));
418            rp.set_index_buffer(
419                self.mesh_cube.index_buf.slice(..),
420                wgpu::IndexFormat::Uint32,
421            );
422            rp.set_vertex_buffer(1, self.instance_buf.slice(..));
423            let inst_count = self.instances.len() as u32;
424            if inst_count > 0 {
425                rp.draw_indexed(0..self.mesh_cube.index_count, 0, 0..inst_count);
426            }
427        }
428
429        self.queue.submit(Some(enc.finish()));
430        frame.present();
431        Ok(())
432    }

446    pub fn render_with<F>(&mut self, f: F) -> Result<()>
447    where
448        F: FnOnce(
449            &wgpu::TextureView,
450            &mut wgpu::CommandEncoder,
451            &wgpu::Device,
452            &wgpu::Queue,
453            (u32, u32),
454        ),
455    {
456        let frame = self.surface.get_current_texture()?;
457        let view = frame
458            .texture
459            .create_view(&wgpu::TextureViewDescriptor::default());
460        let mut enc = self
461            .device
462            .create_command_encoder(&wgpu::CommandEncoderDescriptor {
463                label: Some("encoder"),
464            });
465
466        f(
467            &view,
468            &mut enc,
469            &self.device,
470            &self.queue,
471            self.surface_size(),
472        );
473
474        self.queue.submit(std::iter::once(enc.finish()));
475        frame.present();
476        Ok(())
477    }
