13impl Camera {
14    pub fn view_matrix(&self) -> Mat4 {
15        let dir = Self::dir(self.yaw, self.pitch);
16        Mat4::look_to_rh(self.position, dir, Vec3::Y)
17    }
18
19    pub fn proj_matrix(&self) -> Mat4 {
20        Mat4::perspective_rh(self.fovy, self.aspect.max(0.01), self.znear, self.zfar)
21    }
22
23    pub fn vp(&self) -> Mat4 {
24        self.proj_matrix() * self.view_matrix()
25    }
26
27    fn dir(yaw: f32, pitch: f32) -> Vec3 {
28        let cy = yaw.cos();
29        let sy = yaw.sin();
30        let cp = pitch.cos();
31        let sp = pitch.sin();
32        Vec3::new(cy * cp, sp, sy * cp).normalize()
33    }
34}

49impl CameraController {
50    pub fn new(speed: f32, sensitivity: f32) -> Self {
51        Self {
52            speed,
53            sensitivity,
54            fwd: 0.0,
55            back: 0.0,
56            left: 0.0,
57            right: 0.0,
58            up: 0.0,
59            down: 0.0,
60            dragging: false,
61            last_mouse: None,
62        }
63    }
64
65    pub fn process_keyboard(&mut self, key: winit::keyboard::KeyCode, pressed: bool) {
66        let v = if pressed { 1.0 } else { 0.0 };
67        match key {
68            winit::keyboard::KeyCode::KeyW => self.fwd = v,
69            winit::keyboard::KeyCode::KeyS => self.back = v,
70            winit::keyboard::KeyCode::KeyA => self.left = v,
71            winit::keyboard::KeyCode::KeyD => self.right = v,
72            winit::keyboard::KeyCode::Space => self.up = v,
73            winit::keyboard::KeyCode::ShiftLeft | winit::keyboard::KeyCode::ShiftRight => {
74                self.down = v
75            }
76            _ => {}
77        }
78    }
79
80    pub fn process_mouse_button(&mut self, button: winit::event::MouseButton, pressed: bool) {
81        if button == winit::event::MouseButton::Right {
82            self.dragging = pressed;
83            if !pressed {
84                self.last_mouse = None;
85            }
86        }
87    }
88
89    pub fn process_mouse_move(&mut self, camera: &mut Camera, pos: Vec2) {
90        if self.dragging {
91            if let Some(last) = self.last_mouse {
92                let delta = (pos - last) * self.sensitivity;
93                camera.yaw -= delta.x;
94                camera.pitch -= delta.y;
95                camera.pitch = camera.pitch.clamp(-1.54, 1.54);
96            }
97            self.last_mouse = Some(pos);
98        }
99    }
100
101    pub fn update_camera(&self, camera: &mut Camera, dt: f32) {
102        let dir = super::camera::Camera::dir(camera.yaw, camera.pitch);
103        let right = dir.cross(Vec3::Y).normalize();
104        let up = Vec3::Y;
105
106        let mut vel = Vec3::ZERO;
107        vel += dir * (self.fwd - self.back);
108        vel += right * (self.right - self.left);
109        vel += up * (self.up - self.down);
110        if vel.length_squared() > 0.0 {
111            camera.position += vel.normalize() * self.speed * dt;
112        }
113    }
