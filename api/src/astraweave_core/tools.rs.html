62// simple grid LOS
63pub fn los_clear(
64    obstacles: &std::collections::HashSet<(i32, i32)>,
65    a: crate::IVec2,
66    b: crate::IVec2,
67) -> bool {
68    // Convert schema::IVec2 to glam::IVec2
69    let a_glam = schema_to_glam(a);
70    let b_glam = schema_to_glam(b);
71
72    let mut x = a_glam.x;
73    let mut y = a_glam.y;
74    let dx = (b_glam.x - a_glam.x).signum();
75    let dy = (b_glam.y - a_glam.y).signum();
76    while x != b_glam.x || y != b_glam.y {
77        if obstacles.contains(&(x, y)) {
78            return false;
79        }
80        if x != b_glam.x {
81            x += dx;
82        }
83        if y != b_glam.y {
84            y += dy;
85        }
86    }
87    true
88}

90// 4-neighbor BFS for path existence
91pub fn path_exists(
92    obstacles: &HashSet<(i32, i32)>,
93    start: crate::IVec2,
94    goal: crate::IVec2,
95    bounds: (i32, i32, i32, i32),
96) -> bool {
97    // Convert schema::IVec2 to glam::IVec2
98    let start_glam = schema_to_glam(start);
99    let goal_glam = schema_to_glam(goal);
100
101    use std::collections::VecDeque;
102    let (minx, miny, maxx, maxy) = bounds;
103    let mut q = VecDeque::new();
104    let mut seen = HashSet::new();
105    q.push_back(start_glam);
106    seen.insert((start_glam.x, start_glam.y));
107    while let Some(p) = q.pop_front() {
108        if p.x == goal_glam.x && p.y == goal_glam.y {
109            return true;
110        }
111        for (dx, dy) in [(1, 0), (-1, 0), (0, 1), (0, -1)] {
112            let nx = p.x + dx;
113            let ny = p.y + dy;
114            if nx < minx || ny < miny || nx > maxx || ny > maxy {
115                continue;
116            }
117            if obstacles.contains(&(nx, ny)) {
118                continue;
119            }
120            if seen.insert((nx, ny)) {
121                q.push_back(GlamIVec2::new(nx, ny));
122            }
123        }
124    }
125    false
126}

130// A* grid path (4-neighbor) returning a path (including start & goal) or empty if none.
131pub fn astar_path(
132    obstacles: &std::collections::HashSet<(i32, i32)>,
133    start: crate::IVec2,
134    goal: crate::IVec2,
135    bounds: (i32, i32, i32, i32),
136) -> Vec<crate::IVec2> {
137    // Convert schema::IVec2 to glam::IVec2
138    let start_glam = schema_to_glam(start);
139    let goal_glam = schema_to_glam(goal);
140
141    use std::collections::{BinaryHeap, HashMap, HashSet};
142    #[derive(Copy, Clone, Eq, PartialEq)]
143    struct Node {
144        f: i32,
145        x: i32,
146        y: i32,
147    }
148    impl Ord for Node {
149        fn cmp(&self, other: &Self) -> Ordering {
150            other.f.cmp(&self.f)
151        }
152    }
153    impl PartialOrd for Node {
154        fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
155            Some(self.cmp(other))
156        }
157    }
158
159    let (minx, miny, maxx, maxy) = bounds;
160    let h = |a: GlamIVec2, b: GlamIVec2| (a.x - b.x).abs() + (a.y - b.y).abs();
161    let mut open = BinaryHeap::new();
162    let mut came: HashMap<(i32, i32), (i32, i32)> = HashMap::new();
163    let mut g: HashMap<(i32, i32), i32> = HashMap::new();
164    let mut seen: HashSet<(i32, i32)> = HashSet::new();
165
166    let s = (start_glam.x, start_glam.y);
167    let t = (goal_glam.x, goal_glam.y);
168    g.insert(s, 0);
169    open.push(Node {
170        f: h(start_glam, goal_glam),
171        x: start_glam.x,
172        y: start_glam.y,
173    });
174    seen.insert(s);
175
176    while let Some(Node { f: _, x, y }) = open.pop() {
177        if (x, y) == t {
178            // reconstruct
179            let mut path = vec![glam_to_schema(GlamIVec2::new(x, y))];
180            let mut cur = (x, y);
181            while let Some(&prev) = came.get(&cur) {
182                if prev == cur {
183                    break;
184                }
185                cur = prev;
186                path.push(glam_to_schema(GlamIVec2::new(cur.0, cur.1)));
187            }
188            path.reverse();
189            return path;
190        }
191        let cur_g = *g.get(&(x, y)).unwrap_or(&i32::MAX);
192        for (dx, dy) in [(1, 0), (-1, 0), (0, 1), (0, -1)] {
193            let nx = x + dx;
194            let ny = y + dy;
195            if nx < minx || ny < miny || nx > maxx || ny > maxy {
196                continue;
197            }
198            if obstacles.contains(&(nx, ny)) {
199                continue;
200            }
201            let ng = cur_g + 1;
202            let pos = (nx, ny);
203            if ng < *g.get(&pos).unwrap_or(&i32::MAX) {
204                came.insert(pos, (x, y));
205                g.insert(pos, ng);
206                let prio = ng + h(GlamIVec2::new(nx, ny), goal_glam);
207                open.push(Node {
208                    f: prio,
209                    x: nx,
210                    y: ny,
211                });
212            }
213        }
214    }
215    vec![]
216}

218// Find positions within radius of `from` that have LOS from player but *not* from enemy (crude "cover")
219pub fn find_cover_positions(
220    obstacles: &std::collections::HashSet<(i32, i32)>,
221    bounds: (i32, i32, i32, i32),
222    from: crate::IVec2,
223    player: crate::IVec2,
224    enemy: crate::IVec2,
225    radius: i32,
226) -> Vec<crate::IVec2> {
227    // Convert schema::IVec2 to glam::IVec2
228    let from_glam = schema_to_glam(from);
229    let _player_glam = schema_to_glam(player);
230    let _enemy_glam = schema_to_glam(enemy);
231
232    let (minx, miny, maxx, maxy) = bounds;
233    let mut out = vec![];
234    for dx in -radius..=radius {
235        for dy in -radius..=radius {
236            let nx = from_glam.x + dx;
237            let ny = from_glam.y + dy;
238            if nx < minx || ny < miny || nx > maxx || ny > maxy {
239                continue;
240            }
241            if obstacles.contains(&(nx, ny)) {
242                continue;
243            }
244            let p = GlamIVec2::new(nx, ny);
245            let los_player = los_clear(obstacles, player, glam_to_schema(p));
246            let los_enemy = los_clear(obstacles, enemy, glam_to_schema(p));
247            if los_player && !los_enemy {
248                out.push(glam_to_schema(p));
249            }
250        }
251    }
252    out
