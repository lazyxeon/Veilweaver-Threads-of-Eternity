29#[derive(Default)]
30pub struct World {
31    pub t: f32,
32    pub next_id: Entity,
33    pub obstacles: HashSet<(i32, i32)>,
34    poses: HashMap<Entity, Pose>,
35    health: HashMap<Entity, Health>,
36    team: HashMap<Entity, Team>,
37    ammo: HashMap<Entity, Ammo>,
38    cds: HashMap<Entity, Cooldowns>,
39    names: HashMap<Entity, String>,
40}
41
42impl World {
43    pub fn new() -> Self {
44        Self {
45            t: 0.0,
46            next_id: 1,
47            ..Default::default()
48        }
49    }
50
51    pub fn spawn(&mut self, name: &str, pos: IVec2, team: Team, hp: i32, ammo: i32) -> Entity {
52        let id = self.next_id;
53        self.next_id += 1;
54        self.poses.insert(id, Pose { pos });
55        self.health.insert(id, Health { hp });
56        self.team.insert(id, team);
57        self.ammo.insert(id, Ammo { rounds: ammo });
58        self.cds.insert(
59            id,
60            Cooldowns {
61                map: HashMap::new(),
62            },
63        );
64        self.names.insert(id, name.to_string());
65        id
66    }
67
68    pub fn tick(&mut self, dt: f32) {
69        self.t += dt;
70        for cd in self.cds.values_mut() {
71            for v in cd.map.values_mut() {
72                *v = (*v - dt).max(0.0);
73            }
74        }
75    }
76
77    // getters/setters
78    pub fn pose(&self, e: Entity) -> Option<Pose> {
79        self.poses.get(&e).copied()
80    }
81    pub fn pose_mut(&mut self, e: Entity) -> Option<&mut Pose> {
82        self.poses.get_mut(&e)
83    }
84    pub fn health(&self, e: Entity) -> Option<Health> {
85        self.health.get(&e).copied()
86    }
87    pub fn health_mut(&mut self, e: Entity) -> Option<&mut Health> {
88        self.health.get_mut(&e)
89    }
90    pub fn team(&self, e: Entity) -> Option<Team> {
91        self.team.get(&e).copied()
92    }
93    pub fn ammo(&self, e: Entity) -> Option<Ammo> {
94        self.ammo.get(&e).copied()
95    }
96    pub fn ammo_mut(&mut self, e: Entity) -> Option<&mut Ammo> {
97        self.ammo.get_mut(&e)
98    }
99    pub fn cooldowns(&self, e: Entity) -> Option<&Cooldowns> {
100        self.cds.get(&e)
101    }
102    pub fn cooldowns_mut(&mut self, e: Entity) -> Option<&mut Cooldowns> {
103        self.cds.get_mut(&e)
104    }
105    pub fn name(&self, e: Entity) -> Option<&str> {
106        self.names.get(&e).map(|s| s.as_str())
107    }
108
109    pub fn all_of_team(&self, team_id: u8) -> Vec<Entity> {
110        self.team
111            .iter()
112            .filter_map(|(e, t)| if t.id == team_id { Some(*e) } else { None })
113            .collect()
114    }
115    pub fn enemies_of(&self, team_id: u8) -> Vec<Entity> {
116        self.team
117            .iter()
118            .filter_map(|(e, t)| if t.id != team_id { Some(*e) } else { None })
119            .collect()
120    }
121    pub fn pos_of(&self, e: Entity) -> Option<IVec2> {
122        self.poses.get(&e).map(|p| p.pos)
123    }
124    pub fn obstacle(&self, p: IVec2) -> bool {
125        self.obstacles.contains(&(p.x, p.y))
126    }
